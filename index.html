Milo's Tracker

Track Milo's daily calorie intake and share it between both of you in real time.

Google Sheets setup (one time)

  1. Go to sheets.google.com → create a new spreadsheet
  2. Name it Milo Food Tracker and share with both of you (Editor)
  3. Click Extensions → Apps Script
  4. Delete all existing code, paste the script below, click Save 💾
  5. Click the ▷ Run button on doGet and approve permissions
  6. Click Deploy → New deployment → Web App
  7. Set "Who has access" to Anyone → Deploy → copy the URL
const SHEET_LOG    = 'Log';
const SHEET_DATA   = 'FoodData';
const SHEET_TARGET = 'Targets';

function doGet(e) {
  const ss      = SpreadsheetApp.getActiveSpreadsheet();
  const payload = e && e.parameter && e.parameter.payload;
  const cb      = e && e.parameter && e.parameter.callback;

  let result;
  if (payload) {
    result = handleWrite(ss, JSON.parse(decodeURIComponent(payload)));
  } else {
    result = {
      log:      sheetToObjects(ss, SHEET_LOG),
      foodData: sheetToObjects(ss, SHEET_DATA),
      targets:  sheetToObjects(ss, SHEET_TARGET),
    };
  }

  const json = JSON.stringify(result);
  // JSONP callback — used as fallback when direct fetch is blocked
  if (cb) {
    return ContentService
      .createTextOutput(cb + '(' + json + ')')
      .setMimeType(ContentService.MimeType.JAVASCRIPT);
  }
  // Plain JSON with CORS header so direct fetch works
  return ContentService
    .createTextOutput(json)
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  // Redirect POST to doGet handler for consistency
  return doGet(e);
}

function handleWrite(ss, body) {
  try {
    const action = body.action;

    if (action === 'addLog') {
      const sheet = getOrCreate(ss, SHEET_LOG);
      if (sheet.getLastRow() === 0)
        sheet.appendRow(['Timestamp','Date','User','Food','Qty','Unit','Calories','Notes']);
      sheet.appendRow([
        body.Timestamp || new Date().toISOString(),
        body.Date, body.User, body.Food,
        body.Qty, body.Unit, body.Calories, body.Notes || ''
      ]);

    } else if (action === 'deleteLog') {
      const sheet = getOrCreate(ss, SHEET_LOG);
      const data  = sheet.getDataRange().getValues();
      for (let i = data.length - 1; i >= 1; i--) {
        if (String(data[i][0]) === String(body.Timestamp) &&
            String(data[i][2]) === String(body.User)) {
          sheet.deleteRow(i + 1);
          break;
        }
      }

    } else if (action === 'setFoodData') {
      const sheet = getOrCreate(ss, SHEET_DATA);
      sheet.clearContents();
      sheet.appendRow(['Food','Kcal per 100g','Weight per piece (g)','Kcal per item','Unit','Notes']);
      body.items.forEach(r =>
        sheet.appendRow([r.name, r.kcalPer100g, r.weightPerPiece,
                         r.kcalPerItem, r.unit || 'pieces', r.notes || ''])
      );

    } else if (action === 'setTargets') {
      const sheet = getOrCreate(ss, SHEET_TARGET);
      sheet.clearContents();
      sheet.appendRow(['Food','Target qty','Unit','Kcal']);
      body.targets.forEach(r =>
        sheet.appendRow([r.food, r.qty, r.unit, r.kcal])
      );
      sheet.appendRow(['__daily_target__', body.dailyTarget, 'kcal', '']);
    }

    return { ok: true, action: action };
  } catch(err) {
    return { ok: false, error: err.message };
  }
}

function sheetToObjects(ss, name) {
  const sheet = ss.getSheetByName(name);
  if (!sheet || sheet.getLastRow() < 2) return [];
  const [headers, ...rows] = sheet.getDataRange().getValues();
  return rows.map(r =>
    Object.fromEntries(headers.map((h, i) => [h, r[i]]))
  );
}

function getOrCreate(ss, name) {
  return ss.getSheetByName(name) || ss.insertSheet(name);
}
