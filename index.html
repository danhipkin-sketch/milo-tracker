Milo's Tracker

Track Milo's daily calorie intake and share it between both of you in real time.

One-time Google Sheets setup

  1. Go to sheets.google.com and create a new spreadsheet
  2. Name it Milo Food Tracker
  3. Share it with both of you (Editor access)
  4. Go to Extensions → Apps Script
  5. Paste the script from the How to get the API URL section below and click Deploy
  6. Copy the Web App URL and paste it below

How to get the API URL

  1. In Apps Script, delete all existing code
  2. Paste the code snippet shown after setup (tap "Show Script" below)
  3. Click Deploy → New deployment → Web App
  4. Set "Who has access" to Anyone
  5. Click Deploy and copy the URL
const SHEET_NAME_LOG = 'Log';
const SHEET_NAME_DATA = 'FoodData';
const SHEET_NAME_TARGET = 'Targets';

function doPost(e) {
  const body = JSON.parse(e.postData.contents);
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const action = body.action;

  if (action === 'addLog') {
    const sheet = getOrCreate(ss, SHEET_NAME_LOG);
    if (sheet.getLastRow() === 0) sheet.appendRow(['Timestamp','Date','User','Food','Qty','Unit','Calories','Notes']);
    sheet.appendRow([new Date().toISOString(), body.date, body.user, body.food, body.qty, body.unit, body.calories, body.notes || '']);
  } else if (action === 'deleteLog') {
    const sheet = getOrCreate(ss, SHEET_NAME_LOG);
    const data = sheet.getDataRange().getValues();
    for (let i = data.length - 1; i >= 1; i--) {
      if (data[i][0] === body.timestamp && data[i][2] === body.user) { sheet.deleteRow(i+1); break; }
    }
  } else if (action === 'setFoodData') {
    const sheet = getOrCreate(ss, SHEET_NAME_DATA);
    sheet.clearContents();
    sheet.appendRow(['Food','Kcal per 100g','Weight per piece (g)','Kcal per item','Notes']);
    body.items.forEach(r => sheet.appendRow([r.name, r.kcalPer100g, r.weightPerPiece, r.kcalPerItem, r.notes || '']));
  } else if (action === 'setTargets') {
    const sheet = getOrCreate(ss, SHEET_NAME_TARGET);
    sheet.clearContents();
    sheet.appendRow(['Food','Target qty','Unit','Kcal']);
    body.targets.forEach(r => sheet.appendRow([r.food, r.qty, r.unit, r.kcal]));
    sheet.appendRow(['__daily_target__', body.dailyTarget, 'kcal', '']);
  }

  return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const log = sheetToObjects(ss, SHEET_NAME_LOG);
  const foodData = sheetToObjects(ss, SHEET_NAME_DATA);
  const targets = sheetToObjects(ss, SHEET_NAME_TARGET);
  const json = JSON.stringify({log, foodData, targets});
  const callback = e && e.parameter && e.parameter.callback;
  if (callback) {
    return ContentService.createTextOutput(callback + '(' + json + ')').setMimeType(ContentService.MimeType.JAVASCRIPT);
  }
  return ContentService.createTextOutput(json).setMimeType(ContentService.MimeType.JSON);
}

function sheetToObjects(ss, name) {
  const sheet = ss.getSheetByName(name);
  if (!sheet || sheet.getLastRow() < 2) return [];
  const [headers, ...rows] = sheet.getDataRange().getValues();
  return rows.map(r => Object.fromEntries(headers.map((h,i) => [h, r[i]])));
}

function getOrCreate(ss, name) {
  return ss.getSheetByName(name) || ss.insertSheet(name);
}
